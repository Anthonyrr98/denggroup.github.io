#!/bin/bash

##jobname optional
##partition location 
##(nodes) x (ntasks-per-node) x (cpus-per-task) = (ntasks) = (cpus)


#SBATCH --job-name="jobname"      
#SBATCH--partition=part2          
#SBATCH --nodes=2                 
#SBATCH --ntasks=48
#SBATCH --ntasks-per-node=24
#SBATCH --cpus-per-task=1         


cd /"your .cell .pram files location"   
## e.g /home/rlzhao/DFT/test/  ##e.g /home/rlzhao/DFT/test/

export PATH=/home/rlzhao/BIOVIA/MaterialsStudio20.1/etc/CASTEP/bin/:$PATH 
## 这个位置是 RunCASTEP.sh 的位置 看情况来

RunCASTEP.sh -np 48 "your .cell .pram files name"    
## np 并行运算 48=ntasks e.g RunCASTEP.sh -np 48 ZnO

脚本内容意义

选项含义
-n，-ntasks=number指定要运行的任务数。请求为number个任务分配资源，默认每个任务一个cpu核心，注意n为任务总数，不是每个节点任务数
-c，--cpus-per-task=ncpus告知资源管理系统控制进程，作业的每个任务npcus个cpu核心，若未指定此选项，则默认每个任务分配一个cpu核心
-N，--nodes=minnodes[-maxnodes]请求为作业至少分配minnodes个节点，最大分配maxnodes个节点
--ntasks-­per-­node=<ntasks>
-G, --gpus=[<type>:]<number>gpu申请的类型和数量，多种卡使用“，”分割（ --gpus=tesla:3 ）
--gpus-per-node=[<type>:]<number>每个节点的GPU数量
-p，--partion=partition name指定所需资源使用的分区
-w, --nodelist=<node name list>指定任务运行在某个节点上
-J，--job-name=[name]作业名
-o，--output=[file_name]标准输出文件
-e，--error=[file_name]标准错误文件
-t，--time=[dd-hh:mm:ss]作业最大运行时长
--exclusive独占节点
-mail-type=[type]通知类型，可选 all, fail, end，分别对应全通知、故障通知、结束通知
--mail-user=[mail_address]通知邮箱
--nodelist=[nodes]偏好的作业节点
--exclude=[nodes]避免的作业节点
--depend=[state:job_id]作业依赖
--array=[array_spec]序列作业

squeue 用法

squeue ##显示队列中的作业信息
-A、--­­account=：显示用户的作业信息， 用户以“,”分隔。
-­a、--­­all：显示所有队列中的作业及作业步信息，也显示被配置为对用户组隐藏队列的信息。 
-­r、--­­array：以每行一个作业元素方式显示。 
--­noheader：不显示头信息，即不显示第一行“PARTITION AVAIL TIMELIMIT NODES STATE NODELIST”。­
--­hide：不显示隐藏队列中的作业和作业步信息。此为默认行为，不显示配置为 对用户组隐藏队列的信息。
­--­help：显示帮助信息
-­i、--­­iterate=：以间隔秒方式循环显示信息。 
-­j、--­­jobs=：显示作业号的作业，作业号以“,”分隔。­­
-­l、--­­long：显示更多的作业信息
­-p、--­­partition=：显示特定队列信息，以“,”分隔。
-­q 、­­qos=：显示特定qos的作业和作业步，以“,”分隔。
-o、--­­format=：以特定格式显示信 息。参见 ­O , ­­Format=，采用不同参数。

作业信息-squeue主要输出项
•JOBID：作业号。 
•PARTITION：队列名（分区名）。 
•NAME：作业名。 
•USER：用户名。 
•TIME：已运行时间。 
•ST：状态。
					PD：排队中，PENDING。                 R：运行中，RUNNING。 
					CA：已取消，CANCELLED。               CF：配置中，CONFIGURING。 
					CG：完成中，COMPLETING                CD：已完成，COMPLETED。 
					F：已失败，FAILED。                   TO：超时，TIMEOUT。
					NF：节点失效，NODE FAILURE。          SE：特殊退出状态，SPECIAL EXIT STATE
•NODELIST(REASON)：分配给的节点名列表（原因）：
  AssociationJobLimit：作业达到其最大允许的作业数限制。 
  AssociationResourceLimit：作业达到其最大允许的资源限制。 
  InvalidQOS：作业QOS无效。 
  NodeDown：作业所需的节点宕机。 
  NonZeroExitCode：作业停止时退出代码非零。 
  PartitionDown：作业所需的队列出于DOWN状态。
  PartitionNodeLimit：作业所需的节点超过所用队列当前限制。 
  Priority：作业所需的队列存在高等级作业或预留。 
  QOSJobLimit：作业的QOS达到其最大作业数限制。
  QOSResourceLimit：作业的QOS达到其最大资源限制。 
  ReqNodeNotAvail：作业所需的节点无效，如节点宕机
  Resources：作业等待其所需的资源可用。
